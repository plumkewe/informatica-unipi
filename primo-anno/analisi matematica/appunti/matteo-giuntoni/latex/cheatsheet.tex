		\hline
		\textbf{Zeri} &  \multirowcell{2}{\color{blue} $f: [a, b] \longrightarrow \mathbb{R}$ continua\\ \color{red}$f(a) \cdot f(b) < 0 \Longrightarrow \exists c \in (a, b) : f(c) = 0$ } \\
		\hline
		\textbf{Confronto} & \multirowcell{5}{\color{blue}$A \subset \mathbb{R}$, $x_0 \in Acc(x)$, $f,g: A \to \mathbb{R}$ \\ \color[HTML]{00A64F}$\lim\limits_{x\to x_0}f(x) = l_1 \wedge \lim\limits_{x\to x_0}g(x) = l_2 \wedge $ \\ \color[HTML]{00A64F}$\exists U \text{ intorno } x_0:x \in U \cap A \setminus \{x_0\} $ \\ \color{red}$f(x) \leq g(x) \Longrightarrow l_1 \leq l_2$ \\ \color{red}$f(x) \geq g(x) \Longrightarrow l_1 \geq l_2$}\\
		\hline
		\textbf{Weirstrass} & \multirowcell{7}{
			\color{blue}$a,b \in \overline{\mathbb{R}}$, $f: (a,b) \to \mathbb{R}$ continua:\\
			\color[HTML]{00A64F}$\exists \: \lim\limits_{x \to a}f(x) = l_1 \wedge \exists \lim\limits_{x \to b}f(x) = l_2$\\
			\color{red}$f$ lim. inf. $\Longleftrightarrow$ $l_1 \neq -\infty \wedge l_2 \neq -\infty$ \\
			\color{red}$f$ lim. sup. $\Longleftrightarrow$ $l_1 \neq +\infty \wedge l_2 \neq +\infty$\\
			\color{red}$f$ lim. $\Longleftrightarrow$ $l_1 \in \mathbb{R} \wedge l_2 \in \mathbb{R}$\\
			\color{red}$f$ ha min $\Longleftrightarrow \: \exists x_0 \in (a,b) : f(x_0) \leq min\{l_1, l_2\}$\\
			\color{red}$f$ ha max $\Longleftrightarrow \: \exists x_0 \in (a,b) : f(x_0) \geq max\{l_1, l_2\}$
		}\\
		\hline
		\textbf{Carabinieri} & Se due funzioni hanno lo stesso limite ed una è inferiore all'altra, se esiste una $g(x)$ in mezzo a queste due, avrà lo stesso limite\\
		\hline
		\textbf{Lagrange} & \multirowcell{4}{
			\color{blue}$f: [a,b] \to \mathbb{R}$ continua \\
			\color{blue} $f$ derivabile in $(a,b)$ \\
			\color{red} $\exists c \in (a,b): f'(c) = \frac{f(b) - f(a)}{b - a}$
		}\\
		\hline
		\textbf{De l'Hopital} & \multirowcell{3}{
			\color{blue} $\lim\limits_{x\to a^+}f(x) = \lim\limits_{x\to a^+}g(x) = 0$ oppure \\
			\color{blue} $\lim\limits_{x\to a^+}f(x) = \pm\infty$ e $\lim\limits_{x\to a^+}g(x) = \pm\infty$ \\
			\color[HTML]{00A64F} $g'(x) \neq 0$ in un introno destro di $a$ \\
			\color{red} $\lim\limits_{x\to a^+}\frac{f(x)}{g(x)} = \lim\limits_{x\to a^+}\frac{f'(x)}{g'(x)}$
		}\\
		\hline
		\textbf{o-piccolo} & \color{red}$f(x) = o(g(x)) \Longleftrightarrow \lim_{x \rightarrow x_0} \frac{f(x)}{g(x)} = 0$ \\
		\hline
		\textbf{O-grande} & \multirowcell{5}{
			\color{blue}$A \subset \mathbb{R} \wedge x_0 \in Acc(A) \wedge f,g: A \to \mathbb{R}$ \\
			\color{blue}$U$ intorno di $x_0$ \\
			\color{blue}$\exists M \in \mathbb{R}:$\\
			\color{red}$|f(x)| \geq M \cdot |g(x)| \Longrightarrow f(x) = O(g(x))$ \\
			\color[HTML]{00A64F}$\forall x \in U \cap A \setminus \{x_0\}$
		}\\
		\hline
		\textbf{Infinitesima e parte principale} & \multirowcell{4}{
			\color{blue}$A \subset \mathbb{R} \wedge x_0 \in Acc(A)$\\
			\color{blue}$f,g: A \to \mathbb{R}$ infinitesime per $x\to x_0$ \\
			\color[HTML]{00A64F}$\exists L, \alpha \in \mathbb{R}$ con $L \neq 0$ t.c. $f(x) = L \cdot (g(x))^\alpha + o((g(x))^\alpha)$ \\
			\color{red} $f$ infinitesimo di ordine $\alpha$ rispetto a $g$ \\
			\color{red} $L(g(x))^\alpha$ parte principale
		}\\
	\hline
	\end{tabular}
	\caption{Teoremi e definizioni}
\end{table}
